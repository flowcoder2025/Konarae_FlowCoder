/**
 * ID 기반 텍스트 조회
 * @param id - 텍스트 ID (예: "LID.AUTH.LOGIN.TITLE", "BTN.PRIMARY.SUBMIT")
 * @returns 해당 ID의 텍스트, 없으면 ID 반환
 * @example getText("LID.AUTH.LOGIN.TITLE") // "로그인" 또는 "로그인해요"
 */
export function getText(id: string): string {
  return ID_TEXT[id] ?? id
}
